<script>

const iframe = document.getElementById('ytVideo');

const button = document.getElementById('playButton');


// Prüfen, ob Autoplay mit Ton blockiert wurde (auf Smartphones)

// iOS/Android erlauben Autoplay nur stumm, daher Button anzeigen

function checkAutoplay() {

    // Immer Button anzeigen auf mobilen Geräten

    const isMobile = /iPhone|iPad|iPod|Android/i.test(navigator.userAgent);

    if (isMobile) {

        button.style.display = 'block';

    }

}


// Klick auf Button -> Ton aktivieren

button.addEventListener('click', () => {

    // Neuen iframe mit Ton erstellen

    const src = "https://www.youtube.com/embed/szN1EGHB52s?autoplay=1&mute=0";

    iframe.src = src;

    button.style.display = 'none';

});
